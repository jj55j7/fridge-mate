rules_version = '2';

// Firebase Storage Security Rules for Fridge-Mate
service firebase.storage {
  match /b/{bucket}/o {
    // Allow users to upload and read their own profile photos
    match /users/{userId}/profile.jpg {
      // Allow authenticated users to read any profile photo
      allow read: if request.auth != null;
      
      // Allow users to write only their own profile photo
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow users to upload and read their own food photos
    match /users/{userId}/food/{fileName} {
      // Allow authenticated users to read food photos
      allow read: if request.auth != null;
      
      // Allow users to write only their own food photos
      allow write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
